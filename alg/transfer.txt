перенос данных таблицы в объекты и связи ({
	"таблицы" : "таблица",
	с "полями" : ["ключевое поле1", "поле2", "поле3"],
	со "значениями" : [["запись1 ключевое поле1","запись1 поле2","запись1 поле3"], ["запись2 ключевое поле1","запись2 поле2","запись2 поле3"]]
})
{
	//создание таблицы
	создание объекта "таблица"

	//создание полей
	цикл по "полям" (П)
		создать объект "поле"[П]
			
	//привязка полей к таблице	
	цикл по "полям" (П)
		связать объект "поле"[П] и объект "таблица"

	//привязка полей к ключевому полю
	цикл по "полям" (П)
		если индекс("поле"[П]) != индекс("поле"[1])
			связать объект "поле"[П] и объект "поле"[1]
		
	//создание значений
	цикл по измерению1 записей "значений" (З)
		цикл по измерению2 полей "значений" (П)
			если "значение"[З][П] не пустое
				создать объект "значение"[З][П]

	//привязка значений всего столбца к полю по всем полям
	цикл по измерению1 записей "значений" (З)
		цикл по измерению2 полей "значений" (П)
			если объект "значение"[З][П] существует
				связать объект "значение"[З][П] и объект "поле"[П]
		
	//привязка значений всех полей записи к записи по всем записям
	цикл по измерению1 записей "значений" (З)
		цикл по измерению2 полей "значений" (П)
			если индекс("поле"[П]) != индекс("поле"[1])
				если объект "значение"[З][П] существует
					связать объект "значение"[З][П] и объект "значение"[З][1]
				
}
